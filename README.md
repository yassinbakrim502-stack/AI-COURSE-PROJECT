import sys
from chatterbot import ChatBot
from chatterbot.trainers import ListTrainer

def initialize_bot():
    """
    إعداد البوت وتحديد اسم وقاعدة بيانات خاصة به
    """
    print("[+] جاري إعداد الروبوت وتجهيز قاعدة البيانات...")
    
    # استخدام SQLite لكي يحفظ البوت كل ما يتعلمه في ملف مستقل
    bot = ChatBot(
        'AgadirBot',
        storage_adapter='chatterbot.storage.SQLStorageAdapter',
        database_uri='sqlite:///agadir_bot_db.sqlite3'
    )
    return bot

def train_bot(bot):
    """
    تغذية البوت ببيانات حوارية غنية عن مدينة أكادير
    """
    print("[+] جاري تدريب الروبوت على معالم وثقافة أكادير...")
    trainer = ListTrainer(bot)
    
    # مصفوفة الحوارات (يمكنك إضافة المزيد هنا بسهولة)
    conversations = [
        # الحوار الأول: المقاهي
        "فين كاينة أحسن قهوة فأكادير؟", 
        "كاينة قهوة لا فاخرة فالكرنيش، حدا البحر، وكاينين قهاوي زوينين ف شارع الحسن الثاني.",
        
        # الحوار الثاني: الطقس
        "شنو الجو فأكادير اليوم؟", 
        "الجو فأكادير غالباً مشمس ومعتدل، حيت قريبة للمحيط الأطلسي 🌞.",
        
        # الحوار الثالث: الأنشطة
        "شنو ندير فأكادير؟", 
        "سير لـ سوق الأحد، تمشى فالكورنيش، أو ركب ف التيليريفيك لـ أكادير أوفلا!",
        
        # الحوار الرابع: المعالم
        "شكون هوما أهم معالم أكادير؟",
        "أهم المعالم هي قصبة أكادير أوفلا، وحديقة التماسيح (Crocoparc)، ومارينا أكادير.",
        
        # الحوار الخامس: الأكل
        "فين نقدر ناكل طاجين زوين؟",
        "كاينين مطاعم شعبية واعرة فحي تالبورجت، ولا تقدر تمشي لـ أنزا وتامراغت لـ الحوت الطري والطواجن البلدية.",
        
        # الحوار السادس: مدة الزيارة
        "شحال خاصني ديال الوقت باش نزور أكادير؟",
        "من يومين لـ 3 أيام كافية باش تكتشف المدينة والشاطئ، وإيلا عندك سيمانة تقدر تزور النواحي بحال تغازوت ووادي الجنة."
    ]
    
    trainer.train(conversations)
    print("[+] تم الانتهاء من التدريب بنجاح!")

def start_chat_loop(bot):
    """
    بدء حلقة النقاش المستمرة مع المستخدم
    """
    print("\n" + "="*40)
    print("🤖 مرحبا! أنا بوت أكادير (AgadirBot).")
    print("يمكنك سؤالي عن أي شيء يخص المدينة.")
    print("اكتب 'خروج' أو 'exit' لإغلاق البرنامج.")
    print("="*40 + "\n")
    
    while True:
        try:
            # استقبال المدخلات وتنظيفها من المساحات الزائدة
            user_input = input("أنت: ").strip()
            
            # التحقق من رغبة المستخدم في الخروج
            if user_input.lower() in ['خروج', 'exit', 'quit']:
                print("\n🤖 البوت: تبارك الله عليك! شكراً على وقتك، تلاقاو ف فرصة قادمة إن شاء الله.")
                sys.exit(0)
                
            # تخطي المدخلات الفارغة
            if not user_input:
                continue
                
            # الحصول على الإجابة من النموذج المتدرب
            bot_response = bot.get_response(user_input)
            print(f"🤖 البوت: {bot_response}\n")
            
        except (KeyboardInterrupt, EOFError):
            print("\n\n[-] تم إنهاء البرنامج بشكل مفاجئ.")
            break
        except Exception as e:
            print(f"\n[-] حدث خطأ غير متوقع: {e}\n")

if __name__ == "__main__":
    # تنفيـذ البرنامج خطوة بخطوة
    my_bot = initialize_bot()
    train_bot(my_bot)
    start_chat_loop(my_bot)
# 🤖 AgadirBot - ChatterBot Example

مستودع برمي يحتوي على "بوت شات" ذكي مبني باستخدام مكتبة `ChatterBot` في بايثون، تم تدريبه بالدارجة المغربية للإجابة على الأسئلة المتعلقة بمدينة أكادير المغربية (السياحة، الطقس، الأكل، والمعالم).

## ✨ المميزات (Features)
- **قاعدة بيانات SQLite:** يقوم بحفظ الحوارات والتعلم المستمر وتخزينها محلياً.
- **تفاعل مرن:** يدعم معالجة الأخطاء وأوامر الخروج الذكية.
- **لهجة محلية:** يتحدث ويفهم الدارجة المغربية بطلاقة.
